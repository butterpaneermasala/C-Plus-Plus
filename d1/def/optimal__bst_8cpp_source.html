41int optimalBST(const std::vector<int>& freq) {
42 int n = freq.size();
43 if (n == 0)
44 return 0;
45
46 // dp[i][j] will hold the minimum cost of OBST that can be built from keys i
47 // to j
48 std::vector<std::vector<int>> dp(n, std::vector<int>(n, 0));
49
50 // prefix sum to quickly compute the total frequency from i to j
51 std::vector<int> prefix_sum(n + 1, 0);
52 for (int i = 0; i < n; ++i) {
53 prefix_sum[i + 1] = prefix_sum[i] + freq[i];
54 }
55
56 // Cost of a single key is its frequency (base case)
57 for (int i = 0; i < n; ++i) {
58 dp[i][i] = freq[i];
59 }
60
61 // Build cost for chains of length 2 to n
62 for (int length = 2; length <= n; ++length) {
63 for (int i = 0; i <= n - length; ++i) {
64 int j = i + length - 1;
65 dp[i][j] = INT_MAX;
66
67 int fsum = prefix_sum[j + 1] - prefix_sum[i];
68
69 // Try all keys in [i..j] as root
70 for (int r = i; r <= j; ++r) {
71 int cost = (r > i ? dp[i][r - 1] : 0) +
72 (r < j ? dp[r + 1][j] : 0) + fsum;
73 dp[i][j] = std::min(dp[i][j], cost);
74 }
75 }
76 }
77
78 return dp[0][n - 1];
79}
